cmake_minimum_required(VERSION 3.13) 
project(CFastMath)
set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED)

set(SOURCE_FILES 
      src/main.c
      src/fastmath.c
      src/numbers.c
)

add_executable(CFastMath ${SOURCE_FILES})

target_compile_options(CFastMath PRIVATE 
      -Ofast -fno-trapping-math 
      -flto -funroll-loops 
      -falign-functions=64 
      -falign-loops=64 -march=native
)
